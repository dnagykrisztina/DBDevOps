DEFINER = 'mysql.sys'@'localhost'
TRIGGER sys_config_update_set_user
 BEFORE UPDATE
 ON sys_config
 FOR EACH ROW
BEGIN
    IF @sys.ignore_sys_config_triggers != true AND NEW.set_by IS NULL THEN
        SET NEW.set_by = USER();
    END
