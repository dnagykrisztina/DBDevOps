CREATE VIEW SYSTEM.PRODUCT_PRIVS (
  PRODUCT,
  USERID,
  ATTRIBUTE,
  SCOPE,
  NUMERIC_VALUE,
  CHAR_VALUE,
  DATE_VALUE,
  LONG_VALUE
)
AS
  SELECT PRODUCT,
       USERID,
       ATTRIBUTE,
       SCOPE,
       NUMERIC_VALUE,
       CHAR_VALUE,
       DATE_VALUE,
       LONG_VALUE
  FROM SQLPLUS_PRODUCT_PROFILE
  WHERE USERID = 'PUBLIC'
    OR USER LIKE USERID;
